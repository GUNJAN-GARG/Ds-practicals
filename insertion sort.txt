  Implementation of Insertion sort



#include <stdio.h>

int main()
{

    int a[100],m,n,temp=0,j;
    printf("enter the size of array\n");
    scanf("%d",&m);
    printf("enter elements\n");
    for(int i=0;i<m;i++){
        scanf("%d",&a[i]);
    }
    
    for(int i=1;i<m;i++){
        temp=a[i];
        for(j=i-1;temp<a[j]&&j>=0;j--){
            a[j+1]=a[j];
        }
        a[j+1]=temp;
    }
    
    printf("sorted array is\n");
    for(int i=0;i<m;i++){
        printf("%d ",a[i]);
    }
    
}
