  Implementation of Binary search 





#include<stdio.h>
#include<conio.h>
int main()
{

	int a[100],n;
	
	printf("enter no. of elements\n");
	scanf("%d",&n);
	
	printf("enter elements\n");
	for(int i=0;i<n;i++){
		scanf("%d",&a[i]);
	}
	
	printf("enter search element\n");
	int s;
	scanf("%d",&s);

int first=0;
int last=n-1;
int middle=(first+last)/2;

while(first<=last){
	if(a[middle]<s){
		first=middle+1;
	}
	
	else if(a[middle]==s){
		printf("position:%d",middle+1);
		break;
	}
	else{
		last=middle-1;
	}
	
	middle=(first+last)/2;
}

if(first>last){
	printf("not found");
}
}
